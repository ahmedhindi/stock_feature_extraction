def plotter(data, pattern_name):
    fig, ax = plt.subplots(figsize=(32,12))
    candlestick2_ohlc(ax,data['Open'],data['High'],data['Low'],data['Close'],width=0.6) 
    plt.vlines(data.index[data[pattern_name]],data.Close.min(), data.Close.max(),alpha=0.2)
    plt.title(pattern_name, fontsize=25)
